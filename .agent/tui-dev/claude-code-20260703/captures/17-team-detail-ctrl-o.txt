      132         if seen != len(self.tasks):
      133             raise CycleError("cycle detected")
      134         return batches
      135
      136     def owner_grouped_batches(self) -> list[dict[str, list[str]]]:
      137         """Return parallel batches grouped by owner.
      138
      139         Tasks with no owner are grouped under ``"unowned"``.
      140         """
      141
      142         grouped_batches: list[dict[str, list[str]]] = []
      143         for batch in self.parallel_batches():
      144             grouped: dict[str, list[str]] = defaultdict(list)
      145             for task_id in batch:
      146                 task = self.by_id[task_id]
      147                 grouped[task.owner or "unowned"].append(task_id)
      148             grouped_batches.append(dict(sorted(grouped.items())))
      149         return grouped_batches
      150
      151     def _validate_acyclic(self) -> None:
      152         self.parallel_batches()
